@satterly Sure, I have three separate VMs each running a single instance of Alerta all using the same docker file
version: '3.1'
services:
  alerta:
    container_name: alerta
    image: harbor.example.net/library/alerta-web:latest
    volumes:
      - ./alertad.conf:/app/alertad.conf
      - ./forwarder.py:/venv/lib/python3.7/site-packages/alerta/plugins/forwarder.py
    ports:
      - 8080:8080
    networks:
      - dbnet

  alerta_db:
    image: postgres
    volumes:
      - ./pg-data:/var/lib/postgresql/data
    environment:
      POSTGRES_DB: monitoring
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: postgres
    ports:
      - 5432:5432
    restart: always
    networks:
      - dbnet

networks:
  dbnet:
    driver: bridge
Here is a pastebin with all the alertad.conf configs for each of the nodes https://pastebin.com/c0YxH5JP
