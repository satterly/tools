May 14 2019 10:38 UTC

@guohaolee I fixed a few things with the Docker image last night and have just tested the following very basic docker-compose file that sets the Alerta port to 8005 and it worked without a problem.

version: '3.1'
services:
  web:
    image: alerta/alerta-web
    ports:
      - 8005:8080
    depends_on:
      - db
    environment:
      - DEBUG=1  # remove this line to turn DEBUG off
      - DATABASE_URL=postgres://postgres:postgres@db:5432/monitoring
      - PLUGINS=reject,blackout,normalise,enhance
    restart: always
  db:
    image: postgres
    volumes:
      - ./pg-data:/var/lib/postgresql/data
    environment:
      POSTGRES_DB: monitoring
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: postgres
    restart: always

Please pull the latest alert-web image then try the above and let me know if you have any problems.
