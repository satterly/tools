Here is my alertad.conf file @satterly :

SORT_LIST_BY = 'severity'
SEVERITY_MAP = {
    'hardware': 1,
    'network': 2,
    'server': 3,
    'service': 4,
    'aplikasi': 5,
    'fitur': 6,
    'normal': 0,
    'ok': 0,
    'indeterminate': 9
}

COLOR_MAP = {
    'severity': {
        'hardware': '#b71c1c',
        'network': '#b71c1c',
        'server': '#b71c1c',
        'service': '#b71c1c',
        'aplikasi': '#b71c1c',
        'fitur': '#b71c1c',
        'normal': '#1b5e20',
        'ok': '#1b5e20'
    },
    'text': 'white',
    'highlight': 'grey'
}
COLUMNS = ['severity', 'lastReceiveTime', 'service', 'resource', 'event', 'kategori','value']
DEFAULT_NORMAL_SEVERITY = 'ok'
ADMIN_USERS = 'admin@example.com'
PLUGINS = ['prometheus']

ASI_SEVERITY = [
    'hardware', 'network', 'server', 'service', 'aplikasi', 'fitur', 'normal', 'indeterminate'
]
ASI_QUERIES = [
    {'text': 'Insiden', 'query': [['attributes.kategori', 'Insiden'], ['status', 'open']]},
    {'text': 'Preventif', 'query': [['attributes.kategori', 'Preventif'], ['status', 'open']]},
]
ALERTMANAGER_API_URL = 'http://alertmanager:9093'
ALERTMANAGER_SILENCE_MINUTES = 10
ALERTMANAGER_SILENCE_HOURS = 0
ALERTMANAGER_SILENCE_DAYS = 0
ALERT_TIMEOUT = 0

I use this configuration in alerta v7.4.1, v7.3.0, and v6.8.5
