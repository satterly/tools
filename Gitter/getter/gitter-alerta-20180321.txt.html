import requests


def send_alerta_heartbeat(origin, timeout_days=1, tags=[]):
    myheaders = {}
    myheaders['Authorization'] = 'Key apikey'
    myheaders['Content-type'] = 'application/json'
    mydata = {
        'origin': origin,
        'timeout': 30 * timeout_days,
        'tags': tags
    }

    r = requests.post('http://fqdn:8080/api/heartbeat', json=mydata, headers = myheaders)
    return r

if __name__ == '__main__':
    tags = []
    tags.append("notify_email=false")
    send_alerta_heartbeat('sync_script.py',1,tags)
