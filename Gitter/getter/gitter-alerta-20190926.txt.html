@rmyhren I use the text panel in Grafana with the html mode and with the following content:
<table id="critical_alerts" class="table-panel-table">
<thead>
  <th>&nbsp;</th>
  <th>Alert</th>
  <th>Instance</th>
  <th>Status</th>
  <th>Description</th>
</thead>
<tbody>
</tbody>
</table>

<script>
var criticalInterval;
var displayCriticalAlerts = function(data) {
  var items = [];
  $.each(data.alerts, function(i, a) {
    items.push('<tr><td>' + (i + 1) + '</td><td><a href="/alerta/alert/' + a.id + '" target="_blank">' + a.event + '</a></td><td>' + a.resource + '</td><td>' + a.status + '</td><td>' + a.text + '</td></tr>');
  });

  $('#critical_alerts tbody').html(items.join(''));
}

var refreshCriticalAlerts = function() {
  $.getJSON('/alerts?status!=closed&status!=shelved&severity=critical', {
    'sort-by': 'createTime',
    'reverse': true,
    'api-key': 'api_key_from_alerta'
  }, displayCriticalAlerts);
}

$(function() {
  refreshCriticalAlerts();

  clearInterval(criticalInterval);
  criticalInterval = setInterval(refreshCriticalAlerts, 5000);
});
</script>
