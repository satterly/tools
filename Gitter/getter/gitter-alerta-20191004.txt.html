Oct 04 2019 22:03 UTC
here the systemd, ini and app: 
/usr/lib/systemd/system/alerta.service

[Unit]
Description=uWSGI instance to serve alerta server
After=network.target

[Service]
User=nginx
Group=nginx
WorkingDirectory=/app/alertaio/latest/app
Environment=PATH=/app/alertaio/latest/venv/bin
Environment=ALERTA_SVR_CONF_FILE=/etc/alerta.conf
ExecStart=/app/alertaio/latest/venv/bin/uwsgi --ini alerta.ini
ExecStop=/bin/kill -SIGKILL $MAINPID
Restart=on-failure

[Install]
WantedBy=multi-user.target
/app/alertaio/latest/app/alerta.ini

## managed by puppet ##
[uwsgi]
chdir = /app/alertaio/latest/app
#mount = /api=wsgi.py
mount = /api=../venv/lib/python3.6/site-packages/alerta/app.wsgi
callable = app
manage-script-name = true

master = true
processes = 1
#logger = syslog:alerta
req-logger = file:/app/log/alerta/alerta-req.log,maxsize=104857600
logger = file:/app/log/alerta/alerta-err.log,maxsize=104857600

socket = /tmp/alerta.sock
chmod-socket = 664
uid = nginx
gid = nginx
vacuum = true

die-on-term = true
/app/alertaio/latest/app/../venv/lib/python3.6/site-packages/alerta/app.wsgi

#!/usr/bin/env python

try:
    from alerta import app  # alerta >= 5.0
except Exception:
    from alerta.app import app  # alerta < 5.0
all this served behind nginx, conf: 
server {
  listen                *:5050;

  server_name           api.monitoring-alerta.acmedomain;
  client_body_timeout 30;
  client_max_body_size 25m;

  index  index.html index.htm index.php;

  access_log            /app/log/alerta/api.monitoring-alerta.acmedomain.access.log;
  error_log             /app/log/alerta/api.monitoring-alerta.acmedomain.error.log;

  location /api {

    try_files $uri @api;
  }
  location @api {

    include uwsgi_params;
    uwsgi_pass alerta;
    proxy_set_header Host $host:$server_port; # Host header
    proxy_set_header X-Real-IP $remote_addr; # X-Real-IP header
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for; # X-Forwarded-For
  }
}
upstream alerta {  
  server     unix:///tmp/alerta.sock  fail_timeout=10s;
}
