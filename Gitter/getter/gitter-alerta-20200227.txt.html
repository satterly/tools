Here is the MLM alertad.conf I'm using 
DATABASE_URL = 'postgres://postgres:postgres@alerta_db:5432/monitoring'
PLUGINS = ['remote_ip', 'reject', 'blackout', 'normalise', 'enhance', 'heartbeat', 'forwarder']

SIGNUP_ENABLED = "False"
SECRET_KEY = 'changeme'


# Forwarder Settings
HMAC_AUTH_CREDENTIALS = [
    {
        'id': 'AKID002',
        'key': 'supersecret2',
        'algorithm': 'sha256'
    }
]

X_FORWARDED = 'mid-level-manager'
FWD_DESTINATIONS = [
    ('http://nagios.stage.arin.net:8080', 'AKID001', 'supersecret', ['*'])  # forward everything ie. alerts and actions
]

ALLOWED_ENVIRONMENTS = ['Production', 'Development']
