generate_post_data()
{
        cat <<EOF
    {
        "environment": "$alerts_prod",
        "event": "$EVENT",
        "group": "$system_name",
        "origin": "$HOST",
        "resource": "$HOST",
        "service": [
                "$system_name"
        ],
        "attributes": {
            "noc": $NOC
        },
        "severity": "$SEVERITY",
        "text": "$TEXT",
        "rawData": "@/tmp/opstatus.raw.2956"
    }
EOF
}

curl -s -POST https://alerta.tre.se:4431/api/alert \
        -H "Authorization: Key $alerts_api" \
        -H 'Content-type: application/json' \
        -d "$(generate_post_data)" > /dev/null
