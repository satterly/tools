def parse_notification(notification):
    notification = json.loads(notification.decode('utf-8'))
    resources=notification['MonitoringObjectFullName']
    if ":" in resources:
        resources=resources.split(":")[1]

    return Alert(
            resource=resources,
            event='Tool2',
            environment=str(os.environ.get('ENVIRONMENT')),
            service=['Tool2'],
            severity=severity_change(notification['Severity']),
            group=notification['AlertName'],
            text=notification['Description'],
            #event_type=notification['ResolutionState'],
            raw_data=notification,

        )
