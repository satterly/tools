<button ng-disabled="" type="button" class="btn btn-info" ng-click="clickAll(alerts);"><i class="glyphicon glyphicon-check"></i> Select All</button>

$scope.clickAll = function(alertList) {
  if($scope.bulkAlerts.length == alertList.length) {
      $scope.bulkAlerts = [];
  } else {
      $scope.bulkAlerts = [];
      angular.forEach(alertList, function(alert) {
          $scope.bulkAlerts.push(alert.id);
      });
  }
};
