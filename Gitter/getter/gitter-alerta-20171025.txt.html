that is what you should do, yes
nginx talks to uwsgi over a socket though, not tcp
jsut follow the docs and set it up like that
these are my uwsgi and nginx configs - you'll have to work out the bits that have ansible subst variables: 
server {
    listen {{ alerta_port }};

    location /api { try_files $uri @api; }
    location @api {
        include {{ app_home }}/packages/{{ nginx_package }}/conf/uwsgi_params;
        uwsgi_pass unix:{{ app_home }}/var/run/uwsgi.sock;
        proxy_set_header Host $host:$server_port;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    location / {
        root {{ app_home }}/packages/{{ alerta_web_package }}/app;
    }
}
``` 
[uwsgi]
chdir = {{ app_home }}/var/www
mount = /api=wsgi.py
callable = app
manage-script-name = true

master = true
processes = 5
logto = {{ app_home }}/var/log/uwsgi.log

socket = {{ app_home }}/var/run/uwsgi.sock
chmod-socket = 664
vacuum = true

die-on-term = true
daemonize2 = true
pidfile = {{ app_home }}/var/run/uwsgi.pid
